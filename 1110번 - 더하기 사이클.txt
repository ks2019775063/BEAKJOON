1110번 - 더하기 사이클

#include<stdio.h>

int main()
{
	int n, m[5], count=0, sum=0; //m[1]은 계산 이전의 값을 저장, m[2]는 계산 이후의 값을 저장
	scanf("%d",&n);
	m[1]=n;
	do //무조건 한번 실행
	{
		m[2] = 0; //m[2] 초기화
	
		m[2] += (m[1]%10)*10; //m[1]의 1의 자리를 m[2]의 10의 자리에더함. ex) m[1]이 42의 경우 m[2]에 m[1]의 1의자리인 2를 10의자리로 더하여 m[2]가 20이 된다.
		sum = (m[1]/10) + (m[1]%10); //m[1]의 1의 자리와 10의 자리를 한 자리 수로 보고 더함(값을 sum에 대입).
		m[2] += sum%10; //sum의 1의 자리를 m[2]에 더함.
		count++; //횟수 증가
		m[1]=m[2]; //m[2]의 값을 m[1]에 저장
	}while(n!=m[2]); //처음 값과 계산 후의 값이 같지 않으면 반복
	printf("%d", count);
}
