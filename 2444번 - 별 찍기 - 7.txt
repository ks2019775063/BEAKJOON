#include<stdio.h>

int main()
{
	int i, j, n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		for(j=n-1;j>=i;j--)
		{
			printf(" ");
		}
		for(j=2;j<=i*2;j++)
		{
			printf("*");
		}
		printf("\n");
	}
	for(i=n-1;i>=1;i--)
	{
		for(j=i;j<n;j++)
		{
			printf(" ");
		}
		for(j=i*2;j>=2;j--)
		{
			printf("*");
		}
		printf("\n");
	}
}
