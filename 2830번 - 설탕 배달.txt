#include<stdio.h>

int main()
{
	int n, m, i, j, sum=0, count=0, no=0;
	scanf("%d", &n);
	m=n;
	while(1)
	{
		if(m/5!=0 && (m%5)%3==0)
		{
			count++;
			sum+=5;
			m -= 5;
		}
		else if(m/3!=0)
		{
			count++;
			sum += 3;
			m -=3;
		}
		else
		{
			no++;
		}
		if(n==sum)
		{
			printf("%d", count);
			break;
		}
		else if(no>0)
		{
			printf("-1");
			break;
		}
	}
	
}
