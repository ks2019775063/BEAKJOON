#include<stdio.h>

int num[10005];

int main()
{
	int i, j, n, m, sum;
	for(i=1;i<=9999;i++)
	{
		sum=0;
		m=i;
		n=10000;
		for(j=1;j<=4;j++)
		{
			n/=10;
			if(m<n)
			{
				continue;
			}
			sum += m/n;
			m %= n;
		}
		sum+=i;
		num[sum]++;
	}
	for(i=1;i<=10000;i++)
	{
		if(num[i]==0)
		{
			printf("%d\n",i);
		}
	}
}
